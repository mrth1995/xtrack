<script lang="ts">
	import { CATEGORY_META, type Category } from '$lib/expenses/schemas';
	import { formatIDR } from '$lib/expenses/formatters';

	interface Props {
		open: boolean;
		expense: { id: string; category: string; amount: number } | null;
		onSkip: () => void;
		onSave: (note: string) => void;
	}

	let { open, expense, onSkip, onSave }: Props = $props();

	let noteText = $state('');
	let inputEl: HTMLTextAreaElement | undefined = $state();

	$effect(() => {
		if (open && inputEl) {
			// iOS autofocus workaround (Pitfall 2): defer to next tick so
			// the touch event has cleared.
			setTimeout(() => {
				inputEl?.focus();
			}, 50);
		}
		if (!open) noteText = '';
	});

	function emojiFor(category: string): string {
		return CATEGORY_META[category as Category]?.emoji ?? '';
	}
</script>

	id: string;
	amount: number;
	category: string;
	note: string | null;
	spent_at: string;

/**
 * Compute the WIB (Asia/Jakarta, UTC+7) calendar-day window for "today" and
 * return the corresponding UTC ISO bounds for use against `expenses.spent_at`
 * (timestamptz). Fixes Pitfall 6 in 02-RESEARCH.md — naive UTC bounds cause
 * expenses logged 00:00–07:00 WIB to be misclassified as "yesterday".
 */
function wibTodayBoundsUtc(now: Date = new Date()): { startUtc: string; endUtc: string } {
	// "en-CA" produces YYYY-MM-DD; pin to Asia/Jakarta so the date matches WIB.
	const wibDate = now.toLocaleDateString('en-CA', { timeZone: 'Asia/Jakarta' });
	const startUtc = new Date(`${wibDate}T00:00:00+07:00`).toISOString();
	// End-exclusive: start of the next WIB day.
	const endLocalMidnight = new Date(`${wibDate}T00:00:00+07:00`);
	endLocalMidnight.setUTCDate(endLocalMidnight.getUTCDate() + 1);
	const endUtc = endLocalMidnight.toISOString();
	return { startUtc, endUtc };
}

		throw redirect(303, '/onboarding');
	}

	const supabase = locals.supabase;
	const householdId = locals.householdId;
	const { startUtc, endUtc } = wibTodayBoundsUtc();

		.from('expenses')
